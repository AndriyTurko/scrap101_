function setLocation(ref, confirm = false) {

  if(ref) {
    jQuery.ajax({
      url: '/index.php?route=extension/module/geo_location/set_location',
      type: 'post',
      data: 'ref=' + ref + '&confirm=' + confirm,
      dataType: 'json',
      success: function(json) {
        if(json['city_name']) {
          $('.customer__locate-wrap .current_location').html(json['city_name']);
          if(json['ref']) {
            $('.cities_list li').removeClass('active');
            $('.cities_list li[data-ref="' + json['ref'] + '"]').addClass('active');
            if($('*').is('.module__product_additional')) {
              $('.module__product_additional').find('.module__product_additional-shipping').load('index.php?route=ajax/product/productAdditional', { product_id: $('input[name="product_id"]').val(), sku: $('input[name="sku"]').val(), shipping_only: true });
            }
          }
        }
        $('input[name="search_location"]').val('');
        $('.confirm_locate').hide();
        $('#geo_modal').modal('hide');
      }
    });
  }
}

$('.cities_list, .geo__search-list').on('click', 'li', function() {
  setLocation($(this).attr('data-ref'));
});

let sAjaxSearch;
$(".form-geo_search input").on('keyup', function(e) {
    let mEl = $(this).parents('.form-geo_search');
    clearTimeout(sAjaxSearch);
    sAjaxSearch = setTimeout(function() {
        searchLocation(mEl)
    }, 1000);
});

function searchLocation(mEl) {

    let searchKeyword = mEl.find('input').val().trim();
    let btn = mEl.find('button');
    let btn_text = btn.html();
    let list = mEl.find('.geo__search-list');

    $(document).on('mouseup', function(e) {
        let container = mEl;
        if (container.has(e.target).length === 0) {
            list.slideUp(200);
        }
    });

    if (searchKeyword.length > 1) {
        $.ajax({
            url: '/index.php?route=extension/module/geo_location/search',
            data: 'keyword=' + encodeURIComponent(searchKeyword),
            dataType: 'json',
            method: 'POST',
            beforeSend: function() {
                btn.html('<i class="fa fa-circle-o-notch fa-spin"></i>');
            },
            success: function(json) {
                if (json) {
                    list.html('');
                    let html = '<ul>';
                    
                    if(json['cities']) {
                      json['cities'].forEach(function(el) {
                        html += '<li data-ref="' + el.ref + '" class="list-group-item list-group-item-action align-items-center d-flex search-item justify-content-between">';
                        html += el.city_name;
                        html += '</li>';
                      });
                    } else {
                      html += '<li class="list-group-item list-group-item-action align-items-center d-flex search-item justify-content-between">';
                        html += '<span>не знайдено</span>';
                      html += '</li>';
                    }
                    
                    html += '<ul>';
                    list.append(html);
                    list.slideDown(200);
                }
                btn.html(btn_text);
            },
            error: function() {
                list.html('');
                btn.html(btn_text);
            },
        });
    }
}

// Autocomplete */
(function($) {
	$.fn.Soautocomplete = function(option) {
		return this.each(function() {
			this.timer = null;
			this.items = new Array();

			$.extend(this, option);

			$(this).attr('autocomplete', 'off');

			// Focus
			$(this).on('focus', function() {
				this.request();
			});

			// Blur
			$(this).on('blur', function() {
				setTimeout(function(object) {
					object.hide();
				}, 200, this);
			});

			// Keydown
			$(this).on('keydown', function(event) {
				switch(event.keyCode) {
					case 27: // escape
						this.hide();
						break;
					default:
						this.request();
						break;
				}
			});

			// Click
			this.click = function(event) {
				event.preventDefault();

				value = $(event.target).parent().attr('data-value');

				if (value && this.items[value]) {
					this.select(this.items[value]);
				}
			}

			// Show
			this.show = function() {
				var pos = $(this).position();

				$(this).siblings('ul.dropdown-menu').css({
					top: pos.top + $(this).outerHeight(),
					left: pos.left
				});

				$(this).siblings('ul.dropdown-menu').show();
			}

			// Hide
			this.hide = function() {
				$(this).siblings('ul.dropdown-menu').hide();
			}

			// Request
			this.request = function() {
				clearTimeout(this.timer);

				this.timer = setTimeout(function(object) {
					object.source($(object).val(), $.proxy(object.response, object));
				}, 200, this);
			}

			// Response
			this.response = function(json) {
				html = '';

				if (json.length) {
					for (i = 0; i < json.length; i++) {
						this.items[json[i]['value']] = json[i];
					}

					for (i = 0; i < json.length; i++) {
						if (!json[i]['category']) {
						html += '<li class="media" data-value="' + json[i]['value'] + '" title="' + json[i]['label'] + '">';
						if(json[i]['image'] && json[i]['show_image'] && json[i]['show_image'] == 1 ) {
							html += '	<a class="media-left" href="' + json[i]['link'] + '"><img class="pull-left" src="' + json[i]['image'] + '"></a>';
						}

						html += '<div class="media-body">';
						html += '<a href="' + json[i]['link'] + '" title="' + json[i]['label'] + '"><span>' +json[i]['cate_name'] + json[i]['label'] + '</span></a>';
						if(json[i]['price'] && json[i]['show_price'] && json[i]['show_price'] == 1){
							html += '	<div class="box-price">';
							if (!json[i]['special']) {
								html += '<span class="price">Ціна : '+json[i]['price']+'</span>';;
							} else {
								html += '</span><span class="price-new">' + json[i]['special'] + '</span>'+'<span class="price-old" style="text-decoration:line-through;">' + json[i]['price']  ;
							}
							
							html += '	</div>';
						}
						html += '</div></li>';
						html += '<li class="clearfix"></li>';
						}
					}

					// Get all the ones with a categories
					var category = new Array();

					for (i = 0; i < json.length; i++) {
						if (json[i]['category']) {
							if (!category[json[i]['category']]) {
								category[json[i]['category']] = new Array();
								category[json[i]['category']]['name'] = json[i]['category'];
								category[json[i]['category']]['item'] = new Array();
							}

							category[json[i]['category']]['item'].push(json[i]);
						}
					}

					for (i in category) {
						html += '<li class="dropdown-header">' + category[i]['name'] + '</li>';

						for (j = 0; j < category[i]['item'].length; j++) {
							html += '<li data-value="' + category[i]['item'][j]['value'] + '"><a href="#">&nbsp;&nbsp;&nbsp;' + category[i]['item'][j]['label'] + '</a></li>';
						}
					}
				}

				if (html) {
					this.show();
				} else {
					this.hide();
				}

				$(this).siblings('ul.dropdown-menu').html(html);
			}

			$(this).after('<ul class="dropdown-menu"></ul>');

		});
	}
})(window.jQuery);

$('#review').on('click', '.r_profit button', function() {

	let rate = $(this).attr('data-rate'),
		review_id = $(this).parents('[data-review-id]').attr('data-review-id'),
		parent = $(this).parents('[data-review-id]');

	if(!parent.hasClass('disabled')) {

		$.ajax({
			url: 'index.php?route=product/product/rateReview',
			type: 'post',
			dataType: 'json',
			data: 'rate=' + rate + '&review_id=' + review_id,
			beforeSend: function() {
				parent.addClass('disabled');
				parent.find('.rate_ajax_error').before('<i class="fa fa-circle-o-notch fa-spin"></i>');
			},
			complete: function() {
				parent.removeClass('disabled');
				parent.find('.fa-spin').remove();
			},
			success: function(json) {

				if (json['error']) {
					parent.find('.rate_ajax_error').html(json['error']).stop(true, true).fadeIn(200).delay(2000).fadeOut(200);
				}

				if (json['success']) {
					parent.find('.rate_ajax_error').html(json['success']).stop(true, true).fadeIn(200).delay(2000).fadeOut(200);

					if(json['results']) {
						parent.find('.review-useful .p_count').html(json['results']['useful']);
						parent.find('.review-not-useful .p_count').html(json['results']['not_useful']);
						parent.parents('.reviews__tree-wrap').find('.reviews__tree-useful_procent span').text(json['results']['useful_procent']);
						if(json['results']['useful_procent'] > 0) {
							parent.parents('.reviews__tree-wrap').find('.reviews__tree-useful_procent').show();
						} else {
							parent.parents('.reviews__tree-wrap').find('.reviews__tree-useful_procent').hide();
						}
						
					}
				}
			},
			error: function(jqXHR) {
				parent.find('.rate_ajax_error').html(jqXHR['status'] + ": " + jqXHR['statusText']).stop(true, true).fadeIn(200).delay(2000).fadeOut(200);
			} 
		});
	}
});
